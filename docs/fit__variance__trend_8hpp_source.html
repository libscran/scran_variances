124template<typename Float_>
125void fit_variance_trend(
126 const std::size_t n,
127 const Float_* const mean,
128 const Float_* const variance,
129 Float_* const fitted,
130 Float_* const residuals,
131 FitVarianceTrendWorkspace<Float_>& workspace,
132 const FitVarianceTrendOptions& options)
133{
134 auto& xbuffer = workspace.xbuffer;
135 sanisizer::resize(xbuffer, n);
136 auto& ybuffer = workspace.ybuffer;
137 sanisizer::resize(ybuffer, n);
138
139 const auto quad = [](Float_ x) -> Float_ {
140 return x * x * x * x;
141 };
142
143 std::size_t counter = 0;
144 const Float_ min_mean = options.minimum_mean;
145 for (decltype(I(n)) i = 0; i < n; ++i) {
146 if (!options.mean_filter || mean[i] >= min_mean) {
147 xbuffer[counter] = mean[i];
148 if (options.transform) {
149 ybuffer[counter] = std::pow(variance[i], 0.25); // Using the same quarter-root transform that limma::voom uses.
150 } else {
151 ybuffer[counter] = variance[i];
152 }
153 ++counter;
154 }
155 }
156
157 if (counter < 2) {
158 throw std::runtime_error("not enough observations above the minimum mean");
159 }
160
161 auto& sorter = workspace.sorter;
162 sorter.set(counter, xbuffer.data());
163 auto& work = workspace.sort_workspace;
164 sorter.permute(std::array<Float_*, 2>{ xbuffer.data(), ybuffer.data() }, work);
165
166 WeightedLowess::Options<Float_> smooth_opt;
167 if (options.use_minimum_width) {
168 smooth_opt.span = options.minimum_window_count;
169 smooth_opt.span_as_proportion = false;
170 smooth_opt.minimum_width = options.minimum_width;
171 } else {
172 smooth_opt.span = options.span;
173 }
174 smooth_opt.num_threads = options.num_threads;
175
176 // Using the residual array to store the robustness weights as a placeholder;
177 // we'll be overwriting this later.
178 WeightedLowess::compute(counter, xbuffer.data(), ybuffer.data(), fitted, residuals, smooth_opt);
179
180 // Determining the left edge before we unpermute.
181 const Float_ left_x = xbuffer[0];
182 const Float_ left_fitted = (options.transform ? quad(fitted[0]) : fitted[0]);
183
184 sorter.unpermute(fitted, work);
185
186 // Walking backwards to shift the elements back to their original position
187 // (i.e., before filtering on the mean) on the same array. We need to walk
188 // backwards to ensure that writing to the original position on this array
189 // doesn't clobber the first 'counter' positions containing the fitted
190 // values, at least not until each value is shifted to its original place.
191 for (auto i = n; i > 0; --i) {
192 auto j = i - 1;
193 if (!options.mean_filter || mean[j] >= min_mean) {
194 --counter;
195 fitted[j] = (options.transform ? quad(fitted[counter]) : fitted[counter]);
196 } else {
197 fitted[j] = mean[j] / left_x * left_fitted; // draw a y = x line to the origin from the left of the fitted trend.
198 }
199 }
200
201 for (decltype(I(n)) i = 0; i < n; ++i) {
202 residuals[i] = variance[i] - fitted[i];
203 }
204 return;
205}
