118void fit_variance_trend(size_t n, const Float_* mean, const Float_* variance, Float_* fitted, Float_* residuals, FitVarianceTrendWorkspace<Float_>& workspace, const FitVarianceTrendOptions& options) {
119 auto& xbuffer = workspace.xbuffer;
120 xbuffer.resize(n);
121 auto& ybuffer = workspace.ybuffer;
122 ybuffer.resize(n);
123
124 auto quad = [](Float_ x) -> Float_ {
125 return x * x * x * x;
126 };
127
128 size_t counter = 0;
129 Float_ min_mean = options.minimum_mean;
130 for (size_t i = 0; i < n; ++i) {
131 if (!options.mean_filter || mean[i] >= min_mean) {
132 xbuffer[counter] = mean[i];
133 if (options.transform) {
134 ybuffer[counter] = std::pow(variance[i], 0.25); // Using the same quarter-root transform that limma::voom uses.
135 } else {
136 ybuffer[counter] = variance[i];
137 }
138 ++counter;
139 }
140 }
141
142 if (counter < 2) {
143 throw std::runtime_error("not enough observations above the minimum mean");
144 }
145
146 auto& sorter = workspace.sorter;
147 sorter.set(counter, xbuffer.data());
148 auto& work = workspace.sort_workspace;
149 sorter.permute(std::array<Float_*, 2>{ xbuffer.data(), ybuffer.data() }, work);
150
151 WeightedLowess::Options<Float_> smooth_opt;
152 if (options.use_minimum_width) {
153 smooth_opt.span = options.minimum_window_count;
154 smooth_opt.span_as_proportion = false;
155 smooth_opt.minimum_width = options.minimum_width;
156 } else {
157 smooth_opt.span = options.span;
158 }
159 smooth_opt.num_threads = options.num_threads;
160
161 // Using the residual array to store the robustness weights as a placeholder;
162 // we'll be overwriting this later.
163 WeightedLowess::compute(counter, xbuffer.data(), ybuffer.data(), fitted, residuals, smooth_opt);
164
165 // Determining the left edge before we unpermute.
166 Float_ left_x = xbuffer[0];
167 Float_ left_fitted = (options.transform ? quad(fitted[0]) : fitted[0]);
168
169 sorter.unpermute(fitted, work);
170
171 // Walking backwards to shift the elements back to their original position
172 // (i.e., before filtering on the mean) on the same array. We need to walk
173 // backwards to ensure that writing to the original position on this array
174 // doesn't clobber the first 'counter' positions containing the fitted
175 // values, at least not until each value is shifted to its original place.
176 for (size_t i = n; i > 0; --i) {
177 auto j = i - 1;
178 if (!options.mean_filter || mean[j] >= min_mean) {
179 --counter;
180 fitted[j] = (options.transform ? quad(fitted[counter]) : fitted[counter]);
181 } else {
182 fitted[j] = mean[j] / left_x * left_fitted; // draw a y = x line to the origin from the left of the fitted trend.
183 }
184 }
185
186 for (size_t i = 0; i < n; ++i) {
187 residuals[i] = variance[i] - fitted[i];
188 }
189 return;
190}
